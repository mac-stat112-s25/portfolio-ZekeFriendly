```{r}
mn_wi <- elections_small %>% 
  filter(state_name %in% c("Minnesota", "Wisconsin")) %>% 
  mutate(dem_change = dem_pct_20 - dem_pct_16) %>% 
  arrange(dem_change) %>% 
  select(state_name, county_name, dem_pct_20, dem_pct_16, dem_change)
